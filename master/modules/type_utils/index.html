Utility Concepts#

#include <dplx/cncr/concepts.hpp>
namespace dplx::cncr {}
template<typename T, typename ...Ts>
concept any_of#

Fullfilled if \(\exists U \in Ts .\; T = U\) holds.

template<typename T, typename ...Ts>
concept none_of#

Fullfilled if \(\forall U \in Ts .\; T \neq U\) holds.
